title Build Sequence

participant BuildController
participant Cell
participant Builder

participant GameTable
participant Player
BuildController->Cell:setHeight()
Cell->Builder:isValidBuild()

alt ok

Cell<-Builder:return w/o exceptions

Cell->Cell:this.height = height; ...
BuildController<-Cell:BUILDOK
BuildController->Player:updateTurnState()
BuildController->GameTable:setNews(new News(Cell,Builder),"BUILDOK")
else Generic invalid build
Cell<-Builder:throws InvalidBuildException
BuildController<-Cell:BUILDKO\n
BuildController->GameTable:setNews(new News(Cell,Builder),"BUILDKO")
else God-Specific exceptions
end
